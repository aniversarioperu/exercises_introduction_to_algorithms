\item[2.3-2] Modified merge procedure
\begin{lstlisting}
A = [p, ... q, ... r]

n1 = q - p + 1
n2 = r - q

L = [i for i in A[1:n1]]
R = [i for i in A[1:n2]]

i, j = 1, 1

for k in p..r:
  if L[i] <= R[j]:
    A[k] = L[i]
    if i == n1:
      A[k+1..r] = R[j..n2]
      break
    i += 1

  else:
    A[k] = R[j]
    if j == n2:
      A[k+1..r] = L[i..n1]
      break
    j += 1
\end{lstlisting}
